package org.mphasis.controllers;

import java.io.IOException;
import java.io.PrintWriter;
import java.util.List;

import javax.servlet.RequestDispatcher;
import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import javax.servlet.http.HttpSession;

import org.mphasis.model.Cust;
import org.mphasis.model.User;
import org.mphasis.services.CustService;
import org.mphasis.services.CustServiceImpl;
import org.mphasis.services.UserService;
import org.mphasis.services.UserServiceImpl;


@WebServlet("/login")
public class LoginController extends HttpServlet {
	private static final long serialVersionUID = 1L;

	UserService userService=new UserServiceImpl();
	CustService custService=new CustServiceImpl();
	protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		PrintWriter pw=response.getWriter();
		String username=request.getParameter("username");
		String password=request.getParameter("password");
		boolean b=userService.isAdminAuthenticated(username, password);
		HttpSession session=request.getSession();
		User user=new User();
		
		if(b){
			user.setUsername(username);
			user.setPassword(password);
			user.setType("Admin"); 
			session.setAttribute("admin", user);
			List<Cust> customers=custService.getCustomers();
			request.setAttribute("custs", customers);
			RequestDispatcher rd=request.getRequestDispatcher("home1.jsp");
			rd.forward(request, response);
			
		}
		
		else {
		    RequestDispatcher rd=request.getRequestDispatcher("home.jsp");
			request.setAttribute("msg", "Invalid Credentials");
			rd.forward(request, response);
		}
	
	}}
